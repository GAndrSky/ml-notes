class Bottleneck(nn.Module): expansion = 4 def __init__(self, in_ch, planes, stride=1): super().__init__() self.cv1 = nn.Conv2d(in_ch, planes, 1, bias=False); self.bn1 = nn.BatchNorm2d(planes) self.cv2 = nn.Conv2d(planes, planes, 3, stride=stride, padding=1, bias=False); self.bn2 = nn.BatchNorm2d(planes) self.cv3 = nn.Conv2d(planes, planes*4, 1, bias=False); self.bn3 = nn.BatchNorm2d(planes*4) self.skip = nn.Identity() if stride != 1 or in_ch != planes*4: self.skip = nn.Sequential(nn.Conv2d(in_ch, planes*4, 1, stride=stride, bias=False), nn.BatchNorm2d(planes*4)) def forward(self, x): out = F.relu(self.bn1(self.cv1(x))) out = F.relu(self.bn2(self.cv2(out))) return F.relu(self.bn3(self.cv3(out)) + self.skip(x))

ManualBatchNorm (PyTorch)

class ManualBN(nn.Module): def __init__(self, C, eps=1e-5, momentum=0.1): super().__init__() self.gamma = nn.Parameter(torch.ones(C)) self.beta = nn.Parameter(torch.zeros(C)) self.eps = eps; self.m = momentum self.register_buffer('rm', torch.zeros(C)) self.register_buffer('rv', torch.ones(C)) def forward(self, x): # x: (B, C, H, W) if self.training: mu = x.mean([0,2,3]); var = x.var([0,2,3], unbiased=False) self.rm.mul_(1-self.m).add_(mu.detach()*self.m) self.rv.mul_(1-self.m).add_(var.detach()*self.m) else: mu, var = self.rm, self.rv x = (x - mu[:,None,None]) / (var[:,None,None] + self.eps).sqrt() return self.gamma[:,None,None] * x + self.beta[:,None,None]

BatchNorm NumPy forward + backward

def bn_forward(x, gamma, beta, eps=1e-5): # x: (B, C) mu = x.mean(0); var = x.var(0) xhat = (x - mu) / np.sqrt(var + eps) return gamma * xhat + beta, (x, xhat, mu, var, gamma, eps) def bn_backward(dy, cache): x, xhat, mu, var, gamma, eps = cache; B = x.shape[0] dgamma = (dy * xhat).sum(0); dbeta = dy.sum(0) dxhat = dy * gamma dvar = (dxhat*(x-mu)*(-0.5)*(var+eps)**(-1.5)).sum(0) dmu = (dxhat/(-np.sqrt(var+eps))).sum(0) + dvar*(-2*(x-mu)/B).sum(0) dx = dxhat/np.sqrt(var+eps) + dvar*2*(x-mu)/B + dmu/B return dx, dgamma, dbeta
